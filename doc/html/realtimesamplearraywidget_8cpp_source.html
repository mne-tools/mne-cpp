200 void RealTimeSampleArrayWidget::update(Subject*)
201 {
202  double dValue = 0;
203  double dPositionDifference = 0.0;
204  QVector<double> vecSamples = m_pRTSA->getSampleArray();
205  for(unsigned char i = 0; i < m_pRTSA->getArraySize(); ++i)
206  {
207  dValue = vecSamples[i]*m_fScaleFactor - m_dMiddle;
208  dPositionDifference = m_dPosition - (m_dPosX+ui.m_qFrame->width());
209 
210  if((dPositionDifference >= 0) || m_bStartFlag)
211  {
212  if(m_bStartFlag)
213  dPositionDifference = 0;
214 
215  m_qMutex.lock();
216  m_qPainterPath = QPainterPath();
217  m_dPosition = m_dPosX + dPositionDifference;
218  m_qPainterPath.moveTo(m_dPosition, m_dPosY-dValue);
219  m_qMutex.unlock();
220  m_bStartFlag = false;
221 
222  if(!m_bFrozen)
223  m_pTimeCurrentDisplay->setHMS(m_pTime->hour(),m_pTime->minute(),m_pTime->second(),m_pTime->msec());
224  }
225 
226  else
227  {
228  m_qMutex.lock();
229  m_qPainterPath.lineTo(m_dPosition, m_dPosY-dValue);
230  m_qMutex.unlock();
231  }
232 
233  m_dPosition = m_dPosition + m_dSampleWidth;
234  }
235 }
