102 FiffInfo::SPtr RtDataClient::readInfo()
103 {
104  FiffInfo::SPtr p_pFiffInfo(new FiffInfo());
105  bool t_bReadMeasBlockStart = false;
106  bool t_bReadMeasBlockEnd = false;
107 
108  FiffStream t_fiffStream(this);
109  //
110  // Find the start
111  //
112  FiffTag::SPtr t_pTag;
113  while(!t_bReadMeasBlockStart)
114  {
115  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
116  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_MEAS_INFO)
117  {
118  printf("FIFF_BLOCK_START FIFFB_MEAS_INFO\n");
119  t_bReadMeasBlockStart = true;
120  }
121  }
122 
123  //
124  // Parse until the endblock
125  //
126 
127  bool dev_head_t_read = false;
128  bool ctf_head_t_read = false;
129 
130  while(!t_bReadMeasBlockEnd)
131  {
132  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
133  //
134  // megacq parameters
135  //
136  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_DACQ_PARS)
137  {
138  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_DACQ_PARS)
139  {
140  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
141  if(t_pTag->kind == FIFF_DACQ_PARS)
142  p_pFiffInfo->acq_pars = t_pTag->toString();
143  else if(t_pTag->kind == FIFF_DACQ_STIM)
144  p_pFiffInfo->acq_stim = t_pTag->toString();
145  }
146  }
147  //
148  // Coordinate transformations if the HPI result block was not there
149  //
150  if (t_pTag->kind == FIFF_COORD_TRANS)
151  {
152  if (!dev_head_t_read)
153  {
154  p_pFiffInfo->dev_head_t = t_pTag->toCoordTrans();
155  dev_head_t_read = true;
156  }
157  else if (!ctf_head_t_read)
158  {
159  p_pFiffInfo->ctf_head_t = t_pTag->toCoordTrans();
160  ctf_head_t_read = true;
161  }
162  }
163  //
164  // Polhemus data
165  //
166  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_ISOTRAK)
167  {
168  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_ISOTRAK)
169  {
170  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
171 
172  if(t_pTag->kind == FIFF_DIG_POINT)
173  p_pFiffInfo->dig.append(t_pTag->toDigPoint());
174  }
175  }
176  //
177  // Projectors
178  //
179  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_PROJ)
180  {
181  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_PROJ)
182  {
183  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
184  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_PROJ_ITEM)
185  {
186  FiffProj proj;
187  qint32 countProj = p_pFiffInfo->projs.size();
188  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_PROJ_ITEM)
189  {
190  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
191  switch (t_pTag->kind)
192  {
193  case FIFF_NAME: // First proj -> Proj is created
194  proj = FiffProj();
195  p_pFiffInfo->projs.append(proj);
196  p_pFiffInfo->projs[countProj].desc = t_pTag->toString();
197  break;
198  case FIFF_PROJ_ITEM_KIND:
199  p_pFiffInfo->projs[countProj].kind = *(t_pTag->toInt());
200  break;
201  case FIFF_NCHAN: // First data -> FiffNamedMatrix is created
202 // p_pFiffInfo->projs[countProj].data = FiffNamedMatrix();//obsolete
203  p_pFiffInfo->projs[countProj].data->ncol = *(t_pTag->toInt());
204  break;
205  case FIFF_PROJ_ITEM_NVEC:
206  p_pFiffInfo->projs[countProj].data->nrow = *(t_pTag->toInt());
207  break;
208  case FIFF_MNE_PROJ_ITEM_ACTIVE:
209  p_pFiffInfo->projs[countProj].active = *(t_pTag->toInt());
210  break;
211  case FIFF_PROJ_ITEM_CH_NAME_LIST:
212  p_pFiffInfo->projs[countProj].data->col_names = FiffStream::split_name_list(t_pTag->toString());
213  break;
214  case FIFF_PROJ_ITEM_VECTORS:
215  p_pFiffInfo->projs[countProj].data->data = t_pTag->toFloatMatrix().cast<double>();
216  break;
217  }
218  }
219  }
220  }
221  }
222  // Check consisty
223  for(qint32 i = 0; i < p_pFiffInfo->projs.size(); ++i)
224  {
225  if(p_pFiffInfo->projs[i].data->data.rows() != p_pFiffInfo->projs[i].data->nrow)
226  p_pFiffInfo->projs[i].data->data.transposeInPlace();
227  }
228 
229  //
230  // CTF compensation info
231  //
232  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_MNE_CTF_COMP)
233  {
234  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_MNE_CTF_COMP)
235  {
236  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
237  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_MNE_CTF_COMP_DATA)
238  {
239  FiffCtfComp comp;
240  qint32 countComp = p_pFiffInfo->comps.size();
241  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_MNE_CTF_COMP_DATA)
242  {
243  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
244  switch (t_pTag->kind)
245  {
246  case FIFF_MNE_CTF_COMP_KIND: //First comp -> create comp
247  comp = FiffCtfComp();
248  p_pFiffInfo->comps.append(comp);
249  p_pFiffInfo->comps[countComp].ctfkind = *(t_pTag->toInt());
250  break;
251  case FIFF_MNE_CTF_COMP_CALIBRATED:
252  p_pFiffInfo->comps[countComp].save_calibrated = *(t_pTag->toInt());
253  break;
254 // case FIFF_MNE_CTF_COMP_DATA:
255 // p_pFiffInfo->comps[countComp]->data = *(t_pTag->toNamedMatrix());
256 // break;
257  }
258  }
259  }
260  }
261  }
262  //
263  // Bad channels
264  //
265  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_MNE_BAD_CHANNELS)
266  {
267  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_MNE_BAD_CHANNELS)
268  {
269  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
270  if(t_pTag->kind == FIFF_MNE_CH_NAME_LIST)
271  p_pFiffInfo->bads = FiffStream::split_name_list(t_pTag->data());
272  }
273  }
274  //
275  // General
276  //
277  switch(t_pTag->kind)
278  {
279  case FIFF_SFREQ:
280  p_pFiffInfo->sfreq = *(t_pTag->toFloat());
281  break;
282  case FIFF_HIGHPASS:
283  p_pFiffInfo->highpass = *(t_pTag->toFloat());
284  break;
285  case FIFF_LOWPASS:
286  p_pFiffInfo->lowpass = *(t_pTag->toFloat());
287  break;
288  case FIFF_NCHAN:
289  p_pFiffInfo->nchan = *(t_pTag->toInt());
290  break;
291  case FIFF_MEAS_DATE:
292  p_pFiffInfo->meas_date[0] = t_pTag->toInt()[0];
293  p_pFiffInfo->meas_date[1] = t_pTag->toInt()[1];
294  break;
295  }
296 
297  if (t_pTag->kind == FIFF_CH_INFO)
298  p_pFiffInfo->chs.append(t_pTag->toChInfo());
299 
300  // END MEAS
301  if(t_pTag->kind == FIFF_BLOCK_END && *t_pTag->toInt() == FIFFB_MEAS_INFO)
302  {
303  printf("FIFF_BLOCK_END FIFFB_MEAS_INFO\n");
304  t_bReadMeasBlockEnd = true;
305  }
306  }
307 
308  //
309  // Add the channel information and make a list of channel names
310  // for convenience
311  //
312  for (qint32 c = 0; c < p_pFiffInfo->nchan; ++c)
313  p_pFiffInfo->ch_names << p_pFiffInfo->chs[c].ch_name;
314 
315  return p_pFiffInfo;
316 }
