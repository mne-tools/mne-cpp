343  bool TMSIDriver::getSampleMatrixValue(MatrixXf& sampleMatrix)
344 {
345  //Check if the driver DLL was loaded
346  if(!m_bDllLoaded)
347  return false;
348 
349  //Check if device was initialised and connected correctly
350  if(!m_bInitDeviceSuccess)
351  {
352  cout << "Plugin TMSI - ERROR - getSampleMatrixValue() - Cannot start to get samples from device because device was not initialised correctly" << endl;
353  return false;
354  }
355 
356  sampleMatrix.setZero(); // Clear matrix - set all elements to zero
357  uint iSamplesWrittenToMatrix = 0;
358  int channelMax = 0;
359  int sampleMax = 0;
360  int sampleIterator = 0;
361 
362  //get samples from device until the complete matrix is filled, i.e. the samples per block size is met
363  while(iSamplesWrittenToMatrix < m_uiSamplesPerBlock)
364  {
365  //Get sample block from device
366  LONG ulSizeSamples = m_oFpGetSamples(m_HandleMaster, (PULONG)m_lSignalBuffer, m_lSignalBufferSize);
367  LONG ulNumSamplesReceived = ulSizeSamples/(m_uiNumberOfAvailableChannels*4);
368 
369  //Only do the next steps if there was at least one sample received, otherwise skip and wait until at least one sample was received
370  if(ulNumSamplesReceived > 0)
371  {
372  int actualSamplesWritten = 0; //Holds the number of samples which are actually written to the matrix in this while procedure
373 
374  //Write the received samples to an extra buffer, so that they are not getting lost if too many samples were received. These are then written to the next matrix (block)
375  for(int i=0; i<ulNumSamplesReceived; i++)
376  {
377  for(uint j=i*m_uiNumberOfAvailableChannels; j<(i*m_uiNumberOfAvailableChannels)+m_uiNumberOfChannels; j++)
378  m_vSampleBlockBuffer.push_back((double)m_lSignalBuffer[j]);
379  }
380 
381  //If the number of available channels is smaller than the number defined by the user -> set the channelMax to the smaller number
382  if(m_uiNumberOfAvailableChannels < m_uiNumberOfChannels)
383  channelMax = m_uiNumberOfAvailableChannels;
384  else
385  channelMax = m_uiNumberOfChannels;
386 
387  //If the number of the samples which were already written to the matrix plus the last received number of samples is larger then the defined block size
388  //-> only fill until the matrix is completeley filled with samples. The other (unused) samples are still stored in the vector buffer m_vSampleBlockBuffer and will be used in the next matrix which is to be sent to the circular buffer
389  if(iSamplesWrittenToMatrix + ulNumSamplesReceived > m_uiSamplesPerBlock)
390  sampleMax = m_uiSamplesPerBlock - iSamplesWrittenToMatrix + sampleIterator;
391  else
392  sampleMax = ulNumSamplesReceived + sampleIterator;
393 
394  //Read the needed number of samples from the vector buffer to store them in the matrix
395  for(; sampleIterator < sampleMax; sampleIterator++)
396  {
397  for(int channelIterator = 0; channelIterator < channelMax; channelIterator++)
398  {
399  sampleMatrix(channelIterator, sampleIterator) = ((m_vSampleBlockBuffer.first() * (m_bUseUnitGain ? m_vUnitGain[channelIterator] : 1)) + (m_bUseUnitOffset ? m_vUnitOffSet[channelIterator] : 0)) * (m_bUseChExponent ? pow(10., (double)m_vExponentChannel[channelIterator]) : 1);
400  m_vSampleBlockBuffer.pop_front();
401  }
402 
403  actualSamplesWritten ++;
404  }
405 
406  iSamplesWrittenToMatrix = iSamplesWrittenToMatrix + actualSamplesWritten;
407  }
