616 void FiffTag::convert_matrix_from_file_data(FiffTag* tag)
617 /*
618  * Assumes that the input is in the non-native byte order and needs to be swapped to the other one
619  */
620 {
621  int ndim;
622  int k;
623  int *dimp,*data,kind,np,nz;
624  float *fdata;
625  double *ddata;
626  unsigned int tsize = tag->size();
627 
628  if (fiff_type_fundamental(tag->type) != FIFFTS_FS_MATRIX)
629  return;
630  if (tag->data() == NULL)
631  return;
632  if (tsize < sizeof(fiff_int_t))
633  return;
634 
635  dimp = ((fiff_int_t *)((tag->data())+tag->size()-sizeof(fiff_int_t)));
636  swap_intp(dimp);
637  ndim = *dimp;
638  if (fiff_type_matrix_coding(tag->type) == FIFFTS_MC_DENSE) {
639  if (tsize < (ndim+1)*sizeof(fiff_int_t))
640  return;
641  dimp = dimp - ndim;
642  for (k = 0, np = 1; k < ndim; k++) {
643  swap_intp(dimp+k);
644  np = np*dimp[k];
645  }
646  }
647  else {
648  if (tsize < (ndim+2)*sizeof(fiff_int_t))
649  return;
650  if (ndim > 2) /* Not quite sure what to do */
651  return;
652  dimp = dimp - ndim - 1;
653  for (k = 0; k < ndim+1; k++)
654  swap_intp(dimp+k);
655  nz = dimp[0];
656  if (fiff_type_matrix_coding(tag->type) == FIFFTS_MC_CCS)
657  np = nz + dimp[2] + 1; /* nz + n + 1 */
658  else if (fiff_type_matrix_coding(tag->type) == FIFFTS_MC_RCS)
659  np = nz + dimp[1] + 1; /* nz + m + 1 */
660  else
661  return; /* Don't know what to do */
662  /*
663  * Take care of the indices
664  */

666  swap_intp(data+k);
667  np = nz;
668  }
669  /*
670  * Now convert data...
671  */
672  kind = fiff_type_base(tag->type);
673  if (kind == FIFFT_INT) {
674  for (data = (int *)(tag->data()), k = 0; k < np; k++)
675  swap_intp(data+k);
676  }

684  }
