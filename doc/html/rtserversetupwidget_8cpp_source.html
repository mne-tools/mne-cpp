220 void RtServerSetupWidget::cmdConnectionChanged(bool p_bConnectionStatus)

222  if(p_bConnectionStatus)

225  // set frequency txt
226  //

236  // set connectors
237  //
238  QMap<qint32, QString>::ConstIterator it = m_pRtServer->m_qMapConnectors.begin();

242  {
243  if(this->ui.m_qComboBox_Connector->findData(it.key()) == -1)

245  this->ui.m_qComboBox_Connector->insertItem(idx, it.value(), it.key());
246  ++idx;
247  }
248  else
249  idx = this->ui.m_qComboBox_Connector->findData(it.key()) + 1;
250  }
251  this->ui.m_qComboBox_Connector->setCurrentIndex(this->ui.m_qComboBox_Connector->findData(m_pRtServer->m_iActiveConnectorId));
252 
253  //UI enables/disables
254  this->ui.m_qLabel_ConnectionStatus->setText(QString("Connected"));
255  this->ui.m_qLineEdit_Ip->setEnabled(false);
256  this->ui.m_qPushButton_Connect->setText(QString("Disconnect"));
