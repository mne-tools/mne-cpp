170 MatrixXf& MNEHemisphere::getTriCoords(float p_fScaling)
171 {
172  if(m_TriCoords.size() == 0)
173  {
174  m_TriCoords = MatrixXf(3,3*tris.rows());
175  for(qint32 i = 0; i < tris.rows(); ++i)
176  {
177  m_TriCoords.col(i*3) = rr.row( tris(i,0) ).transpose().cast<float>();
178  m_TriCoords.col(i*3+1) = rr.row( tris(i,1) ).transpose().cast<float>();
179  m_TriCoords.col(i*3+2) = rr.row( tris(i,2) ).transpose().cast<float>();
180  }
181  }
182 
183  m_TriCoords *= p_fScaling;
184 
185  return m_TriCoords;
186 }

191 bool MNEHemisphere::transform_hemisphere_to(fiff_int_t dest, const FiffCoordTrans &p_Trans)
192 {
193  FiffCoordTrans trans(p_Trans);
194 
195  if (this->coord_frame == dest)
196  {
197 // res = src;
198  return true;
199  }
200 
201  if (trans.to == this->coord_frame && trans.from == dest)
202  trans.invert_transform();
203  else if(trans.from != this->coord_frame || trans.to != dest)
204  {
205  printf("Cannot transform the source space using this coordinate transformation");//Consider throw
206  return false;
207  }
208 
209  MatrixXf t = trans.trans.block(0,0,3,4);
210 // res = src;
211  this->coord_frame = dest;
212  MatrixXf t_rr = MatrixXf::Ones(this->np, 4);
213  t_rr.block(0, 0, this->np, 3) = this->rr;
214  MatrixXf t_nn = MatrixXf::Zero(this->np, 4);
215  t_nn.block(0, 0, this->np, 3) = this->nn;
216 
217  this->rr = (t*t_rr.transpose()).transpose();
218  this->nn = (t*t_nn.transpose()).transpose();
219 
220  return true;
221 }
