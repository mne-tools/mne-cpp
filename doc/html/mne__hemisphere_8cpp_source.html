136 bool MNEHemisphere::transform_hemisphere_to(fiff_int_t dest, FiffCoordTrans* trans)
137 {
138  if (this->coord_frame == dest)
139  {
140 // res = src;
141  return false;
142  }
143 
144  if (trans->to == this->coord_frame && trans->from == dest)
145  FiffCoordTrans::invert_transform(trans);
146  else if(trans->from != this->coord_frame || trans->to != dest)
147  {
148  printf("Cannot transform the source space using this coordinate transformation");//Consider throw
149  return false;
150  }
151 
152  MatrixXf t = trans->trans.block(0,0,3,4);
153 // res = src;
154  this->coord_frame = dest;
155  MatrixXf t_rr = MatrixXf::Ones(this->np, 4);
156  t_rr.block(0, 0, this->np, 3) = this->rr;
157  MatrixXf t_nn = MatrixXf::Zero(this->np, 4);
158  t_nn.block(0, 0, this->np, 3) = this->nn;
159 
160  this->rr = (t*t_rr.transpose()).transpose();
161  this->nn = (t*t_nn.transpose()).transpose();
162 
163  return true;
164 }

169 MatrixXf* MNEHemisphere::getTriCoords(float p_fScaling)
170 {
171  if(m_pTriCoords == NULL)
172  {
173  m_pTriCoords = new MatrixXf(3,3*tris.rows());
174  for(qint32 i = 0; i < tris.rows(); ++i)
175  {
176  m_pTriCoords->block(0,i*3,3,1) = rr.row( tris(i,0) ).transpose();
177  m_pTriCoords->block(0,i*3+1,3,1) = rr.row( tris(i,1) ).transpose();
178  m_pTriCoords->block(0,i*3+2,3,1) = rr.row( tris(i,2) ).transpose();
179  }
180  }
181 
182  *m_pTriCoords *= p_fScaling;
183 
184  return m_pTriCoords;
185 }
