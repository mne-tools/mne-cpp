90 void FilterData::designFilter()
91 {
92  switch(m_designMethod) {
93  case Tschebyscheff: {
94  ParksMcClellan filter(m_iFilterOrder, m_dCenterFreq, m_dBandwidth, m_dParksWidth, (ParksMcClellan::TPassType)m_Type);
95  m_dCoeffA = filter.FirCoeff;
96 
97  //fft-transform m_dCoeffA in order to be able to perform frequency-domain filtering
98  fftTransformCoeffs();
99 
100  break;
101  }
102 
103  case Cosine: {
104  CosineFilter filtercos;
105 
106  switch(m_Type) {
107  case LPF:
108  filtercos = CosineFilter (m_iFFTlength,
109  (m_dCenterFreq)*(m_sFreq/2),
110  m_dParksWidth*(m_sFreq/2),
111  (m_dCenterFreq)*(m_sFreq/2),
112  m_dParksWidth*(m_sFreq/2),
113  m_sFreq,
114  (CosineFilter::TPassType)m_Type);
115 
116  break;
117 
118  case HPF:
119  filtercos = CosineFilter (m_iFFTlength,
120  (m_dCenterFreq)*(m_sFreq/2),
121  m_dParksWidth*(m_sFreq/2),
122  (m_dCenterFreq)*(m_sFreq/2),
123  m_dParksWidth*(m_sFreq/2),
124  m_sFreq,
125  (CosineFilter::TPassType)m_Type);
126 
127  break;
128 
129  case BPF:
130  filtercos = CosineFilter (m_iFFTlength,
131  (m_dCenterFreq + m_dBandwidth/2)*(m_sFreq/2),
132  m_dParksWidth*(m_sFreq/2),
133  (m_dCenterFreq - m_dBandwidth/2)*(m_sFreq/2),
134  m_dParksWidth*(m_sFreq/2),
135  m_sFreq,
136  (CosineFilter::TPassType)m_Type);
137 
138  break;
139  }
140 
141  //This filter is designed in the frequency domain, hence the time domain impulse response need to be shortend by the users dependent number of taps
142  m_dCoeffA.resize(m_iFilterOrder);
143  m_dCoeffA.head(m_iFilterOrder/2) = filtercos.m_dCoeffA.tail(m_iFilterOrder/2);
144  m_dCoeffA.tail(m_iFilterOrder/2) = filtercos.m_dCoeffA.head(m_iFilterOrder/2);
145 
146  //Now generate the fft version of the shortened impulse response
147  fftTransformCoeffs();
148 
149  break;
150  }
151  }
152 
153  switch(m_Type) {
154  case LPF:
155  m_dLowpassFreq = 0;
156  m_dHighpassFreq = m_dCenterFreq*(m_sFreq/2);
157  break;
158 
159  case HPF:
160  m_dLowpassFreq = m_dCenterFreq*(m_sFreq/2);
161  m_dHighpassFreq = 0;
162  break;
163 
164  case BPF:
165  m_dLowpassFreq = (m_dCenterFreq + m_dBandwidth/2)*(m_sFreq/2);
166  m_dHighpassFreq = (m_dCenterFreq - m_dBandwidth/2)*(m_sFreq/2);
167  break;
168  }

175 void FilterData::fftTransformCoeffs()
176 {
177  //zero-pad m_dCoeffA to m_iFFTlength
178  RowVectorXd t_coeffAzeroPad = RowVectorXd::Zero(m_iFFTlength);

193 RowVectorXd FilterData::applyConvFilter(const RowVectorXd& data, bool keepOverhead, CompensateEdgeEffects compensateEdgeEffects) const
194 {
195  if(data.cols()<m_dCoeffA.cols() && compensateEdgeEffects==MirrorData){
196  qDebug()<<QString("Error in FilterData: Number of filter taps(%1) bigger then data size(%2). Not enough data to perform mirroring!").arg(m_dCoeffA.cols()).arg(data.cols());
197  return data;
198  }
199 
200  //Do zero padding or mirroring depending on user input
201  RowVectorXd t_dataZeroPad = RowVectorXd::Zero(2*m_dCoeffA.cols() + data.cols());
202  RowVectorXd t_filteredTime = RowVectorXd::Zero(2*m_dCoeffA.cols() + data.cols());
203 
204  switch(compensateEdgeEffects) {
205  case MirrorData:
206  t_dataZeroPad.head(m_dCoeffA.cols()) = data.head(m_dCoeffA.cols()).reverse(); //front
207  t_dataZeroPad.segment(m_dCoeffA.cols(), data.cols()) = data; //middle
208  t_dataZeroPad.tail(m_dCoeffA.cols()) = data.tail(m_dCoeffA.cols()).reverse(); //back
209  break;
210 
211  case ZeroPad:
212  t_dataZeroPad.segment(m_dCoeffA.cols(), data.cols()) = data;
213  break;
214 
215  default:
216  t_dataZeroPad.segment(m_dCoeffA.cols(), data.cols()) = data;
217  break;
218  }
219 
220  //Do the convolution
221  for(int i=m_dCoeffA.cols(); i<t_filteredTime.cols(); i++)
222  t_filteredTime(i-m_dCoeffA.cols()) = t_dataZeroPad.segment(i-m_dCoeffA.cols(),m_dCoeffA.cols()) * m_dCoeffA.transpose();
223 
224  //Return filtered data
225  if(!keepOverhead)
226  return t_filteredTime.segment(m_dCoeffA.cols()/2, data.cols());
227 
228  return t_filteredTime.head(data.cols()+m_dCoeffA.cols());
229 }

234 RowVectorXd FilterData::applyFFTFilter(const RowVectorXd& data, bool keepOverhead, CompensateEdgeEffects compensateEdgeEffects) const
235 {
236  if(data.cols()<m_dCoeffA.cols() && compensateEdgeEffects==MirrorData) {
237  qDebug()<<QString("Error in FilterData: Number of filter taps(%1) bigger then data size(%2). Not enough data to perform mirroring!").arg(m_dCoeffA.cols()).arg(data.cols());
238  return data;
239  }
240 
241 // std::cout<<"m_iFFTlength: "<<m_iFFTlength<<std::endl;
242 // std::cout<<"2*m_dCoeffA.cols() + data.cols(): "<<2*m_dCoeffA.cols() + data.cols()<<std::endl;
243 
244  if(2*m_dCoeffA.cols() + data.cols()>m_iFFTlength) {
245  qDebug()<<"Error in FilterData: Number of mirroring/zeropadding size plus data size is bigger then fft length!";
246  return data;
247  }
248 
249  //Do zero padding or mirroring depending on user input
250  RowVectorXd t_dataZeroPad = RowVectorXd::Zero(m_iFFTlength);
251 
252  switch(compensateEdgeEffects) {
253  case MirrorData:
254  t_dataZeroPad.head(m_dCoeffA.cols()) = data.head(m_dCoeffA.cols()).reverse(); //front
255  t_dataZeroPad.segment(m_dCoeffA.cols(), data.cols()) = data; //middle
256  t_dataZeroPad.tail(m_dCoeffA.cols()) = data.tail(m_dCoeffA.cols()).reverse(); //back
257  break;
258 
259  case ZeroPad:
260  t_dataZeroPad.head(data.cols()) = data;
261  break;
262 
263  default:
264  t_dataZeroPad.head(data.cols()) = data;
265  break;
266  }
267 
268  //generate fft object
269  Eigen::FFT<double> fft;
270  fft.SetFlag(fft.HalfSpectrum);
271 
272  //fft-transform data sequence
273  RowVectorXcd t_freqData;
274  fft.fwd(t_freqData,t_dataZeroPad);
275 
276  //perform frequency-domain filtering
277  RowVectorXcd t_filteredFreq = m_dFFTCoeffA.array()*t_freqData.array();
278 
279  //inverse-FFT
280  RowVectorXd t_filteredTime;
281  fft.inv(t_filteredTime,t_filteredFreq);
282 
283  //Return filtered data
284  if(!keepOverhead)
285  return t_filteredTime.segment(m_dCoeffA.cols()/2, data.cols());
286 
287  return t_filteredTime.head(data.cols()+m_dCoeffA.cols());
288 }
