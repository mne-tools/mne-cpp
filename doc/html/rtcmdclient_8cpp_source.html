105 void RtCmdClient::sendCommandJSON(const Command &p_command)
106 {
107  const QString t_sCommand = QString("{\"commands\":{%1}}\n").arg(
108  p_command.toStringReadySend());
109 
110  QString t_sReply;
111 
112  if (this->state() == QAbstractSocket::ConnectedState)
113  {
114  qDebug() << "Request: " << t_sCommand;
115 
116  // Send request
117  this->write(t_sCommand.toUtf8().constData(), t_sCommand.size());
118  this->waitForBytesWritten();
119 
120  // Receive response
121  bool respComplete = false;
122  QByteArray t_qByteArrayRaw;
123  do
124  {
125  if (this->waitForReadyRead(100))
126  {
127  t_qByteArrayRaw += this->readAll();
128  // We need a break condition,
129  // because we do not have a stop character and do not know how many bytes to receive.
130  respComplete = t_qByteArrayRaw.count('{')
131  == t_qByteArrayRaw.count('}');
132  }
133  qDebug() << "Response: " << t_qByteArrayRaw.size() << " bytes";
134  } while (!respComplete);
135  t_sReply = QString(t_qByteArrayRaw);
136  }
137  else
138  {
139  qWarning() << "Request was not send, because client is not connected!";
140  }
141 
142  m_qMutex.lock();
143 // m_sAvailableData.append(t_sReply); //ToDo check this
144  m_sAvailableData = t_sReply;
145  m_qMutex.unlock();
146 
147  emit response(t_sReply);
148 }

310 bool RtCmdClient::waitForDataAvailable(qint32 msecs) const
311 {
312  if (m_sAvailableData.size() > 0)
313  return true;
314 
315  qint64 t_msecsStart = QDateTime::currentMSecsSinceEpoch();
316 
317  while (msecs == -1
