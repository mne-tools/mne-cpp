437  std::vector<qint32> t_vfirsti;
438  t_vfirsti.push_back(current);

440 
441  for(quint32 i = 0; i < t_vIndn.size(); ++i)
442  {
443  if (t_vIndn[current].first != t_vIndn[i].first)
444  {
445  current = i;
446  t_vlasti.push_back(i-1);
447  t_vfirsti.push_back(current);
448  }
449  }
450  t_vlasti.push_back(t_vIndn.size()-1);
451 
452

454  {
455  std::vector<int> t_v;
456 
457  for(int l = t_vfirsti[k]; l <= t_vlasti[k]; ++l)
458  t_v.push_back(t_vIndn[l].second);
