191 bool ECGSimulator::start()
192 {
193  //Check if the thread is already or still running. This can happen if the start button is pressed immediately after the stop button was pressed. In this case the stopping process is not finished yet but the start process is initiated.
194  if(this->isRunning())
195  QThread::wait();
196 
197  initChannels();
198 
199  // Start threads
200  m_pECGProducer->start();
201 
202  if(m_pECGProducer->isRunning())
203  {
204  m_bIsRunning = true;
205  QThread::start();
206  return true;
207  }
208  else
209  {
210  qWarning() << "Plugin TMSI - ERROR - TMSIProducer thread could not be started - Either the device is turned off (check your OS device manager) or the driver DLL (TMSiSDK.dll / TMSiSDK32bit.dll) is not installed in the system32 / SysWOW64 directory" << endl;
211  return false;
212  }
213 }
