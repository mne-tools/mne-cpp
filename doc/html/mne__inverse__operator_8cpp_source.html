900 MNEInverseOperator MNEInverseOperator::prepare_inverse_operator(qint32 nave ,float lambda2, bool dSPM, bool sLORETA) const
901 {
902  if(nave <= 0)
903  {
904  printf("The number of averages should be positive\n");
905  return MNEInverseOperator();
906  }
907  printf("Preparing the inverse operator for use...\n");
908  MNEInverseOperator inv(*this);
909  //
910  // Scale some of the stuff
911  //
912  float scale = ((float)inv.nave)/((float)nave);
913  inv.noise_cov->data *= scale;
914  inv.noise_cov->eig *= scale;
915  inv.source_cov->data *= scale;
916  //
917  if (inv.eigen_leads_weighted)
918  inv.eigen_leads->data *= sqrt(scale);
919  //
920  printf("\tScaled noise and source covariance from nave = %d to nave = %d\n",inv.nave,nave);
921  inv.nave = nave;
922  //
923  // Create the diagonal matrix for computing the regularized inverse
924  //
925  VectorXd tmp = inv.sing.cwiseProduct(inv.sing) + VectorXd::Constant(inv.sing.size(), lambda2);
926 // if(inv.reginv)
927 // delete inv.reginv;
928  inv.reginv = VectorXd(inv.sing.cwiseQuotient(tmp));
929  printf("\tCreated the regularized inverter\n");
930  //
931  // Create the projection operator
932  //
933 
934  qint32 ncomp = FiffProj::make_projector(inv.projs, inv.noise_cov->names, inv.proj);
935  if (ncomp > 0)
936  printf("\tCreated an SSP operator (subspace dimension = %d)\n",ncomp);
937 
938  //
939  // Create the whitener
940  //
941 // if(inv.whitener)
942 // delete inv.whitener;
943  inv.whitener = MatrixXd::Zero(inv.noise_cov->dim, inv.noise_cov->dim);
944 
945  qint32 nnzero, k;
946  if (inv.noise_cov->diag == 0)
947  {
948  //
949  // Omit the zeroes due to projection
950  //
951  nnzero = 0;
952 
953  for (k = ncomp; k < inv.noise_cov->dim; ++k)
954  {
955  if (inv.noise_cov->eig[k] > 0)
956  {
957  inv.whitener(k,k) = 1.0/sqrt(inv.noise_cov->eig[k]);
958  ++nnzero;
959  }
960  }
961  //
962  // Rows of eigvec are the eigenvectors
963  //
964  inv.whitener *= inv.noise_cov->eigvec;
965  printf("\tCreated the whitener using a full noise covariance matrix (%d small eigenvalues omitted)\n", inv.noise_cov->dim - nnzero);
966  }
967  else
968  {
969  //
970  // No need to omit the zeroes due to projection
971  //
972  for (k = 0; k < inv.noise_cov->dim; ++k)
973  inv.whitener(k,k) = 1.0/sqrt(inv.noise_cov->data(k,0));
974 
975  printf("\tCreated the whitener using a diagonal noise covariance matrix (%d small eigenvalues discarded)\n",ncomp);
976  }
977  //
978  // Finally, compute the noise-normalization factors
979  //
980  if (dSPM || sLORETA)
981  {
982  VectorXd noise_norm = VectorXd::Zero(inv.eigen_leads->nrow);
983  VectorXd noise_weight;
984  if (dSPM)
985  {
986  printf("\tComputing noise-normalization factors (dSPM)...");
987  noise_weight = VectorXd(inv.reginv);
988  }
989  else
990  {
991  printf("\tComputing noise-normalization factors (sLORETA)...");
992  VectorXd tmp = (VectorXd::Constant(inv.sing.size(), 1) + inv.sing.cwiseProduct(inv.sing)/lambda2);
993  noise_weight = inv.reginv.cwiseProduct(tmp.cwiseSqrt());
994  }
995  VectorXd one;
996  if (inv.eigen_leads_weighted)
997  {
998  for (k = 0; k < inv.eigen_leads->nrow; ++k)
999  {
1000  one = inv.eigen_leads->data.block(k,0,1,inv.eigen_leads->data.cols()).cwiseProduct(noise_weight);
1001  noise_norm[k] = sqrt(one.dot(one));
1002  }
1003  }
1004  else
1005  {
1006 // qDebug() << 32;
1007  double c;
1008  for (k = 0; k < inv.eigen_leads->nrow; ++k)
1009  {
1010 // qDebug() << 321;
1011  c = sqrt(inv.source_cov->data(k,0));
1012 // qDebug() << 322;
1013 // qDebug() << "inv.eigen_leads->data" << inv.eigen_leads->data.rows() << "x" << inv.eigen_leads->data.cols();
1014 // qDebug() << "noise_weight" << noise_weight.rows() << "x" << noise_weight.cols();
1015  one = c*(inv.eigen_leads->data.row(k).transpose()).cwiseProduct(noise_weight);//ToDo eigenleads data -> pointer
1016  noise_norm[k] = sqrt(one.dot(one));
1017 // qDebug() << 324;
1018  }
1019  }
1020 
1021 // qDebug() << 4;
1022 
1023  //
1024  // Compute the final result
1025  //
1026  VectorXd noise_norm_new;
1027  if (inv.source_ori == FIFFV_MNE_FREE_ORI)
1028  {
1029  //
1030  // The three-component case is a little bit more involved
1031  // The variances at three consequtive entries must be squeared and
1032  // added together
1033  //
1034  // Even in this case return only one noise-normalization factor
1035  // per source location
1036  //
1037  VectorXd* t = MNEMath::combine_xyz(noise_norm.transpose());
1038  noise_norm_new = t->cwiseSqrt();//double otherwise values are getting too small
1039  delete t;
1040  //
1041  // This would replicate the same value on three consequtive
1042  // entries
1043  //
1044  // noise_norm = kron(sqrt(mne_combine_xyz(noise_norm)),ones(3,1));
1045  }
1046  VectorXd vOnes = VectorXd::Ones(noise_norm_new.size());
1047  VectorXd tmp = vOnes.cwiseQuotient(noise_norm_new.cwiseAbs());
1048 // if(inv.noisenorm)
1049 // delete inv.noisenorm;
1050 
1051  typedef Eigen::Triplet<double> T;
1052  std::vector<T> tripletList;
1053  tripletList.reserve(noise_norm_new.size());
1054  for(qint32 i = 0; i < noise_norm_new.size(); ++i)
1055  tripletList.push_back(T(i, i, tmp[i]));
1056 
1057  inv.noisenorm = SparseMatrix<double>(noise_norm_new.size(),noise_norm_new.size());
1058  inv.noisenorm.setFromTriplets(tripletList.begin(), tripletList.end());
1059 
1060  printf("[done]\n");
1061  }
1062  else
1063  {
1064 // if(inv.noisenorm)
1065 // delete inv.noisenorm;
1066  inv.noisenorm = SparseMatrix<double>();
1067  }
1068 
1069  return inv;
1070 }
