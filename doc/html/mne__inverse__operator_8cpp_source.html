564 MNEInverseOperator MNEInverseOperator::prepare_inverse_operator(qint32 nave ,float lambda2, bool dSPM, bool sLORETA) const
565 {
566  if(nave <= 0)
567  {
568  printf("The number of averages should be positive\n");
569  return MNEInverseOperator();
570  }
571  printf("Preparing the inverse operator for use...\n");
572  MNEInverseOperator inv(*this);
573  //
574  // Scale some of the stuff
575  //
576  float scale = ((float)inv.nave)/((float)nave);
577  inv.noise_cov->data *= scale;
578  inv.noise_cov->eig *= scale;
579  inv.source_cov->data *= scale;
580  //
581  if (inv.eigen_leads_weighted)
582  inv.eigen_leads->data *= sqrt(scale);
583  //
584  printf("\tScaled noise and source covariance from nave = %d to nave = %d\n",inv.nave,nave);
585  inv.nave = nave;
586  //
587  // Create the diagonal matrix for computing the regularized inverse
588  //
589  VectorXd tmp = inv.sing.cwiseProduct(inv.sing) + VectorXd::Constant(inv.sing.size(), lambda2);
590 // if(inv.reginv)
591 // delete inv.reginv;
592  inv.reginv = VectorXd(inv.sing.cwiseQuotient(tmp));
593  printf("\tCreated the regularized inverter\n");
594  //
595  // Create the projection operator
596  //
597 
598  qint32 ncomp = FiffProj::make_projector(inv.projs, inv.noise_cov->names, inv.proj);
599  if (ncomp > 0)
600  printf("\tCreated an SSP operator (subspace dimension = %d)\n",ncomp);
601 
602  //
603  // Create the whitener
604  //
605 // if(inv.whitener)
606 // delete inv.whitener;
607  inv.whitener = MatrixXd::Zero(inv.noise_cov->dim, inv.noise_cov->dim);
608 
609  qint32 nnzero, k;
610  if (inv.noise_cov->diag == 0)
611  {
612  //
613  // Omit the zeroes due to projection
614  //
615  nnzero = 0;
616 
617  for (k = ncomp; k < inv.noise_cov->dim; ++k)
618  {
619  if (inv.noise_cov->eig[k] > 0)
620  {
621  inv.whitener(k,k) = 1.0/sqrt(inv.noise_cov->eig[k]);
622  ++nnzero;
623  }
624  }
625  //
626  // Rows of eigvec are the eigenvectors
627  //
628  inv.whitener *= inv.noise_cov->eigvec;
629  printf("\tCreated the whitener using a full noise covariance matrix (%d small eigenvalues omitted)\n", inv.noise_cov->dim - nnzero);
630  }
631  else
632  {
633  //
634  // No need to omit the zeroes due to projection
635  //
636  for (k = 0; k < inv.noise_cov->dim; ++k)
637  inv.whitener(k,k) = 1.0/sqrt(inv.noise_cov->data(k,0));
638 
639  printf("\tCreated the whitener using a diagonal noise covariance matrix (%d small eigenvalues discarded)\n",ncomp);
640  }
641  //
642  // Finally, compute the noise-normalization factors
643  //
644  if (dSPM || sLORETA)
645  {
646  VectorXd noise_norm = VectorXd::Zero(inv.eigen_leads->nrow);
647  VectorXd noise_weight;
648  if (dSPM)
649  {
650  printf("\tComputing noise-normalization factors (dSPM)...");
651  noise_weight = VectorXd(inv.reginv);
652  }
653  else
654  {
655  printf("\tComputing noise-normalization factors (sLORETA)...");
656  VectorXd tmp = (VectorXd::Constant(inv.sing.size(), 1) + inv.sing.cwiseProduct(inv.sing)/lambda2);
657  noise_weight = inv.reginv.cwiseProduct(tmp.cwiseSqrt());
658  }
659  VectorXd one;
660  if (inv.eigen_leads_weighted)
661  {
662  for (k = 0; k < inv.eigen_leads->nrow; ++k)
663  {
664  one = inv.eigen_leads->data.block(k,0,1,inv.eigen_leads->data.cols()).cwiseProduct(noise_weight);
665  noise_norm[k] = sqrt(one.dot(one));
666  }
667  }
668  else
669  {
670 // qDebug() << 32;
671  double c;
672  for (k = 0; k < inv.eigen_leads->nrow; ++k)
673  {
674 // qDebug() << 321;
675  c = sqrt(inv.source_cov->data(k,0));
676 // qDebug() << 322;
677 // qDebug() << "inv.eigen_leads->data" << inv.eigen_leads->data.rows() << "x" << inv.eigen_leads->data.cols();
678 // qDebug() << "noise_weight" << noise_weight.rows() << "x" << noise_weight.cols();
679  one = c*(inv.eigen_leads->data.row(k).transpose()).cwiseProduct(noise_weight);//ToDo eigenleads data -> pointer
680  noise_norm[k] = sqrt(one.dot(one));
681 // qDebug() << 324;
682  }
683  }
684 
685 // qDebug() << 4;
686 
687  //
688  // Compute the final result
689  //
690  VectorXd noise_norm_new;
691  if (inv.source_ori == FIFFV_MNE_FREE_ORI)
692  {
693  //
694  // The three-component case is a little bit more involved
695  // The variances at three consequtive entries must be squeared and
696  // added together
697  //
698  // Even in this case return only one noise-normalization factor
699  // per source location
700  //
701  VectorXd* t = MNEMath::combine_xyz(noise_norm.transpose());
702  noise_norm_new = t->cwiseSqrt();//double otherwise values are getting too small
703  delete t;
704  //
705  // This would replicate the same value on three consequtive
706  // entries
707  //
708  // noise_norm = kron(sqrt(mne_combine_xyz(noise_norm)),ones(3,1));
709  }
710  VectorXd vOnes = VectorXd::Ones(noise_norm_new.size());
711  VectorXd tmp = vOnes.cwiseQuotient(noise_norm_new.cwiseAbs());
712 // if(inv.noisenorm)
713 // delete inv.noisenorm;
714 
715  typedef Eigen::Triplet<double> T;
716  std::vector<T> tripletList;
717  tripletList.reserve(noise_norm_new.size());
718  for(qint32 i = 0; i < noise_norm_new.size(); ++i)
719  tripletList.push_back(T(i, i, tmp[i]));
720 
721  inv.noisenorm = SparseMatrix<double>(noise_norm_new.size(),noise_norm_new.size());
722  inv.noisenorm.setFromTriplets(tripletList.begin(), tripletList.end());
723 
724  printf("[done]\n");
725  }
726  else
727  {
728 // if(inv.noisenorm)
729 // delete inv.noisenorm;
730  inv.noisenorm = SparseMatrix<double>();
731  }
732 
733  return inv;

812  printf("Source orientation information not found\n");

874  {
875  printf("\tSource covariance matrix read.\n");
876  }
877  else
878  {
879  printf("\tError: Not able to read source covariance matrix.\n");

881  }

883  // Read the various priors
884  //
885  if(t_pStream->read_cov(*invs, FIFFV_MNE_ORIENT_PRIOR_COV, *inv.orient_prior.data()))
886  {
