370 void NewRealTimeMultiSampleArrayWidget::paintEvent(QPaintEvent*)
371 {
372  QPainter painter(this);
373 
374 
375 // //*************************************************************************************************************
376 // //=============================================================================================================
377 // // Draw white background
378 // //=============================================================================================================
379 //
380 // painter.setBrush(Qt::white);
381 // painter.drawRect(0, 0, width(), height());
382 
383 
384  painter.setPen(QPen(Qt::gray, 1, Qt::DashLine));
385 
386  //*************************************************************************************************************
387  //=============================================================================================================
388  // Draw grid in X direction (each 100ms)
389  //=============================================================================================================
390 
391  double dNumPixelsX = m_pRTMSA_New->getSamplingRate()/10.0f;
392  double dMinMaxDifference = static_cast<double>(m_pRTMSA_New->chInfo()[0].getMaxValue()-m_pRTMSA_New->chInfo()[0].getMinValue());
393  double dActualPosX = 0.0;
394  unsigned short usNumOfGridsX = (unsigned short)(ui.m_qFrame->width()/dNumPixelsX);
395  unsigned short usPosY = ui.m_qFrame->pos().y()+1;
396  unsigned short usPosX = ui.m_qFrame->pos().x()+1;
397  unsigned short usHeight = ui.m_qFrame->height()-2;
398  unsigned short usWidth = ui.m_qFrame->width()-2;
399 
400  for(unsigned short i = 1; i <= usNumOfGridsX; ++i)
401  {
402  dActualPosX = m_dPosX+i*dNumPixelsX;
403  painter.drawLine((int)dActualPosX, usPosY, (int)dActualPosX, usPosY+usHeight);
404  }
405 
406 
407  //*************************************************************************************************************
408  //=============================================================================================================
409  // Draw grid in Y direction
410  //=============================================================================================================
411 
412  double exponent = (int)floor(log10(dMinMaxDifference))-1;//math.h
413  double dim = pow(10.0, exponent);//respectively at 0.001; 0.01, 0.1, 1, 10, 100
414 
415  int NumOfLines = (int)floor(dMinMaxDifference/(dim*5));
416 
417  double dDifferenceToFirstLine = (m_pRTMSA_New->chInfo()[0].getMaxValue()-floor(m_pRTMSA_New->chInfo()[0].getMaxValue()/dim)*dim);
418 
419  double dNumPixelsY = usHeight/NumOfLines;//10.0f;
420  double dActualPosY = usPosY + dDifferenceToFirstLine * (usHeight/dMinMaxDifference);
421 
422  for(unsigned char i = 1; i <= NumOfLines; ++i)
423  {
424  painter.drawLine((int)m_dPosX, (int)dActualPosY, usWidth, (int)dActualPosY);
425  dActualPosY += dNumPixelsY;
426  }
427 
428  //Paint middle value
429 // painter.setPen(QPen(Qt::gray, 1, Qt::SolidLine));
430 // painter.drawText(usWidth-75, usHeight/2, tr("%1%2").arg(m_dMiddle, 0, 'f', 2).arg(m_pRTSM->getUnit()));
431 // painter.setPen(QPen(Qt::gray, 1, Qt::DotLine));
432 // painter.drawLine(m_dPosX, usHeight/2, usWidth, usHeight/2);
433 
434  painter.setPen(QPen(Qt::darkBlue, 1, Qt::SolidLine));
435  painter.setRenderHint(QPainter::Antialiasing);
436 
437 
438  //*************************************************************************************************************
439  //=============================================================================================================
440  // Draw real time curve respectively frozen curve
441  //=============================================================================================================
442 
443  painter.save();
444  qint32 t_iDist = ui.m_qFrame->height() / (m_uiNumChannels+1);
445 
446  if(m_bFrozen)
447  {
448  painter.setPen(QPen(Qt::darkGray, 1, Qt::SolidLine));
449 
450  for(quint32 k = 0; k < m_uiNumChannels; ++k)
451  {
452  painter.translate(0, t_iDist);
453  painter.drawPolyline(m_qVecPolygonF_Freeze[k]);
454  painter.drawText(0, -t_iDist/2, m_pRTMSA_New->chInfo()[m_uiFirstChannel+k].getChannelName());
455  }
456  }
457  else
458  {
459  m_qMutex.lock();
460  for(quint32 k = 0; k < m_uiNumChannels; ++k)
461  {
462  painter.translate(0, t_iDist);
463  painter.drawPolyline(m_qVecPolygonF[k]);
464  painter.drawText(0, -t_iDist/2, m_pRTMSA_New->chInfo()[m_uiFirstChannel+k].getChannelName());
465  }
466  m_qMutex.unlock();
467  }
468 
469  painter.restore();
470 
471  //*************************************************************************************************************
472  //=============================================================================================================
473  // Calculates zoom with the help of new minimum/maximum factors.
474  //=============================================================================================================
475 
476  if(m_bScaling)
477  {
478  int iStartX = m_qPointMouseStartPosition.x();
479 
480  int iEndY = m_qPointMouseEndPosition.y();
481  int iStartY = m_qPointMouseStartPosition.y();
482 
483  // Compute pixel difference
484  int iPixelDifferenceY = abs(iStartY - iEndY);
485 
486  double scale = (m_dMaxValue_init-m_dMinValue_init)/usHeight;
487 
488  if(iStartY>iEndY)
489  {
490  double changeValue = scale * iPixelDifferenceY;
491 
492  if(changeValue*2 < m_dMaxValue_init - m_dMinValue_init)
493  {
494  minValueChanged(m_dMinValue_init + changeValue);
495  maxValueChanged(m_dMaxValue_init - changeValue);
496  }
497  else
498  {
499  double maxChange = (m_dMaxValue_init - m_dMinValue_init)*0.499999;
500  minValueChanged(m_dMinValue_init + maxChange);
501  maxValueChanged(m_dMaxValue_init - maxChange);
502  }
503  }
504  else
505  {
506  double changeValue = scale * iPixelDifferenceY*10;
507 
508  minValueChanged(m_dMinValue_init - changeValue);
509  maxValueChanged(m_dMaxValue_init + changeValue);
510  }
511 
512  double factor = (m_dMaxValue_init-m_dMinValue_init)/(m_pRTMSA_New->chInfo()[0].getMaxValue()-m_pRTMSA_New->chInfo()[0].getMinValue());
513  // Draw text
514  painter.setPen(QPen(Qt::darkCyan, 1, Qt::SolidLine));
515  painter.drawText(iStartX+8, iEndY, tr("Zoom %1x").arg(factor, 0, 'f', 2));
516 
517  }

557  //=============================================================================================================
