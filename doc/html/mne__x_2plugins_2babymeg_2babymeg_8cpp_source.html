191 void BabyMEG::startRecordingFile()
192 {
193  //Setup writing to file
194  if(m_bWriteToFile)
195  {
196  m_pOutfid->finish_writing_raw();
197  m_bWriteToFile = false;
198  m_pTimerRecordingChange->stop();
199  m_pActionRecordFile->setIcon(QIcon(":/images/record.png"));
200  }
201  else
202  {
203  if(!m_pFiffInfo)
204  {
205  QMessageBox msgBox;
206  msgBox.setText("FiffInfo missing!");
207  msgBox.exec();
208  return;
209  }
210 
211  //Initiate the stream for writing to the fif file
212  m_qFileOut.setFileName(m_sRecordFile);
213  if(m_qFileOut.exists())
214  {
215  QMessageBox msgBox;
216  msgBox.setText("The file you want to write already exists.");
217  msgBox.setInformativeText("Do you want to overwrite this file?");
218  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
219  int ret = msgBox.exec();
220  if(ret == QMessageBox::No)
221  return;
222  }
223 
224  m_pOutfid = Fiff::start_writing_raw(m_qFileOut, *m_pFiffInfo, m_cals);
225  fiff_int_t first = 0;
226  m_pOutfid->write_int(FIFF_FIRST_SAMPLE, &first);
227

229 
230  m_pTimerRecordingChange = QSharedPointer<QTimer>(new QTimer);

239 void BabyMEG::setFiffData(QByteArray DATA)
240 {
241  //get the first byte -- the data format
242  int dformat = DATA.left(1).toInt();
243 
244  DATA.remove(0,1);
245  qint32 rows = m_pFiffInfo->nchan;
246  qint32 cols = (DATA.size()/dformat)/rows;
247 
248  qDebug() << "[BabyMEG] Matrix " << rows << "x" << cols << " [Data bytes:" << dformat << "]";
249 
250  MatrixXf rawData(Map<MatrixXf>( (float*)DATA.data(),rows, cols ));
251 
252  for(qint32 i = 0; i < rows*cols; ++i)
253  IOUtils::swap_floatp(rawData.data()+i);
254 
255 
256  if(m_bIsRunning)
257  {
258  if(!m_pRawMatrixBuffer)
259  m_pRawMatrixBuffer = CircularMatrixBuffer<float>::SPtr(new CircularMatrixBuffer<float>(40, rows, cols));
260 
261  m_pRawMatrixBuffer->push(&rawData);
262  }
263  else
264  {
265 // std::cout << "Data coming" << std::endl; //"first ten elements \n" << rawData.block(0,0,1,10) << std::endl;
266 
267  emit DataToSquidCtrlGUI(rawData);
268  }
269 }

381 void BabyMEG::run()
382 {
383 
384  MatrixXf matValue;
385 
386  while(m_bIsRunning)
387  {
388  if(m_pRawMatrixBuffer)
389  {
390  //pop matrix
391  matValue = m_pRawMatrixBuffer->pop();
392 
393  //Write raw data to fif file
394  if(m_bWriteToFile)
395  m_pOutfid->write_raw_buffer(matValue.cast<double>(), m_cals);
396 
397 
398  if(m_pRTMSABabyMEG)
399  {
400 // std::cout << "matValue" << matValue.block(0,0,2,2) << std::endl;
401  //emit values
402  for(qint32 i = 0; i < matValue.cols(); ++i)
403  m_pRTMSABabyMEG->data()->setValue(matValue.col(i).cast<double>());
404  }
405  }
406  }
407 
408  //Close the fif output stream
409  if(m_bWriteToFile)
410  {
411  m_pOutfid->finish_writing_raw();
412  m_bWriteToFile = false;
413  }
414 }
