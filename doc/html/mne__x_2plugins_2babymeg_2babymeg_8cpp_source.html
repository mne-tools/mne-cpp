189 void BabyMEG::startRecordingFile()
190 {
191  //Setup writing to file
192  if(m_bWriteToFile)
193  {
194  m_pOutfid->finish_writing_raw();
195  m_bWriteToFile = false;
196  m_pTimerRecordingChange->stop();
197  m_pActionRecordFile->setIcon(QIcon(":/images/record.png"));
198  }
199  else
200  {
201  if(!m_pFiffInfo)
202  {
203  QMessageBox msgBox;
204  msgBox.setText("FiffInfo missing!");
205  msgBox.exec();
206  return;
207  }
208 
209  //Initiate the stream for writing to the fif file
210  m_qFileOut.setFileName(m_sRecordFile);
211  if(m_qFileOut.exists())
212  {
213  QMessageBox msgBox;
214  msgBox.setText("The file you want to write already exists.");
215  msgBox.setInformativeText("Do you want to overwrite this file?");
216  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
217  int ret = msgBox.exec();
218  if(ret == QMessageBox::No)
219  return;
220  }
221 
222  m_pOutfid = Fiff::start_writing_raw(m_qFileOut, *m_pFiffInfo, m_cals);
223  fiff_int_t first = 0;
224  m_pOutfid->write_int(FIFF_FIRST_SAMPLE, &first);
225 
226  m_bWriteToFile = true;
227 
228  m_pTimerRecordingChange = QSharedPointer<QTimer>(new QTimer);
229  connect(m_pTimerRecordingChange.data(), &QTimer::timeout, this, &BabyMEG::changeRecordingButton);
230  m_pTimerRecordingChange->start(500);
231  }

237 void BabyMEG::setFiffData(QByteArray DATA)
238 {
239  //get the first byte -- the data format
240  int dformat = DATA.left(1).toInt();
241 
242  DATA.remove(0,1);
243  qint32 rows = m_pFiffInfo->nchan;
244  qint32 cols = (DATA.size()/dformat)/rows;
245 
246  qDebug() << "[BabyMEG] Matrix " << rows << "x" << cols << " [Data bytes:" << dformat << "]";
247 
248  MatrixXf rawData(Map<MatrixXf>( (float*)DATA.data(),rows, cols ));
249 
250  for(qint32 i = 0; i < rows*cols; ++i)
251  IOUtils::swap_floatp(rawData.data()+i);
252 
253 
254  if(m_bIsRunning)
255  {
256  if(!m_pRawMatrixBuffer)
257  m_pRawMatrixBuffer = CircularMatrixBuffer<float>::SPtr(new CircularMatrixBuffer<float>(40, rows, cols));
258 
259  m_pRawMatrixBuffer->push(&rawData);
260  }
261  else
262  {
263 // std::cout << "Data coming" << std::endl; //"first ten elements \n" << rawData.block(0,0,1,10) << std::endl;
264 
265  emit DataToSquidCtrlGUI(rawData);
266  }
267 }

379 void BabyMEG::run()
380 {
381 
382  MatrixXf matValue;
383 
384  while(m_bIsRunning)
385  {
386  if(m_pRawMatrixBuffer)
387  {
388  //pop matrix
389  matValue = m_pRawMatrixBuffer->pop();
390 
391  //Write raw data to fif file
392  if(m_bWriteToFile)
393  m_pOutfid->write_raw_buffer(matValue.cast<double>(), m_cals);
394 
395  if(m_pRTMSABabyMEG)
396  {
397  //emit values
398  for(qint32 i = 0; i < matValue.cols(); ++i)
399  m_pRTMSABabyMEG->data()->setValue(matValue.col(i).cast<double>());
400  }
401  }
402  }
403 
404  //Close the fif output stream
405  if(m_bWriteToFile)
406  {
407  m_pOutfid->finish_writing_raw();
408  m_bWriteToFile = false;
409  }
410 }
