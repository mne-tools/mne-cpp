174 QJsonObject Command::toJsonObject() const
175 {
176  QJsonObject p_jsonCommandObject;
177  p_jsonCommandObject.insert("description", QJsonValue(m_sDescription));
178 
179  QJsonObject t_jsonAllParametersObject;
