356 template< typename T>
357 VectorXi MNEMath::sort(Matrix<T, Dynamic, 1> &v, bool desc)
358 {
359  std::vector< std::pair<int,T> > t_vecIdxValue;
360  VectorXi idx(v.size());
361 
362  if(v.size() > 0)
363  {
364  //fill temporal vector
365  for(qint32 i = 0; i < v.size(); ++i)
366  t_vecIdxValue.push_back(std::pair<int,T>(i, v[i]));
367 
368  //sort temporal vector
369  if(desc)
370  std::sort(t_vecIdxValue.begin(), t_vecIdxValue.end(), MNEMath::compareIdxValuePairBiggerThan<T>);
371  else
372  std::sort(t_vecIdxValue.begin(), t_vecIdxValue.end(), MNEMath::compareIdxValuePairSmallerThan<T>);
373 
374  //store results
375  for(qint32 i = 0; i < v.size(); ++i)
376  {
377  idx[i] = t_vecIdxValue[i].first;
378  v[i] = t_vecIdxValue[i].second;
379  }
380  }
381 
382  return idx;
383 }
