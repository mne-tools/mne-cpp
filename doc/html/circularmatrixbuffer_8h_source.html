237 {
238  if(!m_bPause)
239  {
240  unsigned int t_size = pMatrix->size();
241  if(t_size == m_uiRows*m_uiCols)
242  {
243  m_pFreeElements->acquire(t_size);
244  for(unsigned int i = 0; i < t_size; ++i)
245  m_pBuffer[mapIndex(m_iCurrentWriteIndex)] = pMatrix->data()[i];
