488 void MainWindow::updatePluginWidget(IPlugin::SPtr pPlugin)
489 {
490  m_qListDynamicPluginActions.clear();
491  m_qListDynamicDisplayActions.clear();
492 
493  // Add Dynamic Plugin Actions
494  m_qListDynamicPluginActions.append(pPlugin->getPluginActions());
495 
496  m_sCurPluginName = pPlugin->getName();
497 
498  //Garbage collecting
499  if(m_pRunWidget)
500  {
501  delete m_pRunWidget;
502  m_pRunWidget = NULL;
503  }
504 
505  if(pPlugin.isNull())
506  {
507  QWidget* pWidget = new QWidget;
508  setCentralWidget(pWidget);
509  }
510  else
511  {
512  if(!m_bIsRunning)
513  setCentralWidget(pPlugin->setupWidget());
514  else
515  {
516  m_pRunWidget = new RunWidget( m_pDisplayManager->show(pPlugin->getOutputConnectors(), m_pTime, m_qListDynamicDisplayActions));
517 
518  m_pRunWidget->show();
519 
520  if(m_bDisplayMax)//ToDo send events to main window
521  {
522  m_pRunWidget->showFullScreen();
523  connect(m_pRunWidget, &RunWidget::displayClosed, this, &MainWindow::toggleDisplayMax);
524  }
525  else
526  setCentralWidget(m_pRunWidget);
527  }
528  }
529 
530  this->createToolBars();

545 void MainWindow::writeToLog(const QString& logMsg, LogKind lgknd, LogLevel lglvl)
546 {
547  if(lglvl<=m_eLogLevelCurrent)
548  {
549  if(lgknd == _LogKndError)
550  {
551  m_pTextBrowser_Log->insertHtml("<font color=red><b>Error:</b> "+logMsg+"</font>");
552 
553  }
554  else if(lgknd == _LogKndWarning)
555  {
556  m_pTextBrowser_Log->insertHtml("<font color=blue><b>Warning:</b> "+logMsg+"</font>");
557  }
558  else
559  {
560  m_pTextBrowser_Log->insertHtml(logMsg);
561  }
562  m_pTextBrowser_Log->insertPlainText("\n"); // new line
563  //scroll down to the newest entry
564  QTextCursor c = m_pTextBrowser_Log->textCursor();
565  c.movePosition(QTextCursor::End);
566  m_pTextBrowser_Log->setTextCursor(c);
567 
568  m_pTextBrowser_Log->verticalScrollBar()->setValue(m_pTextBrowser_Log->verticalScrollBar()->maximum());
569  }
