161  pick(k) = 1;
162  }
163  else if (kind == FIFFV_EEG_CH && eeg)
164  pick(k) = 1;

166  pick(k) = 1;

168

171  QStringList myinclude;
172  for(k = 0; k < this->nchan; ++k)

179  }

185  myinclude << include[k];
186  ++p;
187  }
188  }

211 {
212  RowVectorXi sel = RowVectorXi::Zero(ch_names.size());
213 
214  QStringList t_includedSelection;
215 
216  qint32 count = 0;
