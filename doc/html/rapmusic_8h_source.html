373 inline int RapMusic::getRank(const MatrixXT& p_matSigma)
374 {
375  int t_iRank;
376  //if once a singularvalue is smaller than epsilon = 10^-5 the following values are also smaller
377  // -> because Singular values are ordered
378  for(t_iRank = p_matSigma.rows()-1; t_iRank > 0; t_iRank--)
379  if (p_matSigma(t_iRank, t_iRank) > 0.00001)
380  break;
381 
382  t_iRank++;//rank corresponding to epsilon
383 
384  return t_iRank;
385 }
