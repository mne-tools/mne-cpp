364 void RawModel::applyFilter(QModelIndexList selected, ParksMcClellan::TPassType type) //ToDo: make this method more efficient
365 {
366  //generate fft object
367  Eigen::FFT<double> fft;
368  fft.SetFlag(fft.HalfSpectrum);
369 
370  for(qint32 i=0; i < selected.size(); ++i) { //iterate through selected channels to filter
371  RowVectorXcd t_freqData;
372 
373  for(qint8 j=0; j < m_data.size(); ++j) { //iterate through m_data list in order to filter them all, ToDo: dont filter already filtered blocks -> check it somehow
374  RowVectorXd t_procData = RowVectorXd::Zero(m_iWindowSize+m_iFilterTaps);
375 
376  //insert part of m_data row and pad the rest of the row with zeros
377  t_procData.block(0,0,1,m_iWindowSize) = m_data[j].row(selected[i].row());
378 
379  fft.fwd(t_freqData,t_procData);
380 
381  //frequency-domain filtering by multiplication
382  RowVectorXcd t_filteredFreq = m_mapFiltersFreq[type].array()*t_freqData.array();
383 
384  //inverse-FFT
385  RowVectorXd t_filteredTime;
386  fft.inv(t_filteredTime,t_filteredFreq);
387 
388  //cuts off ends at front and end
389  m_procData[j].row(selected[i].row()) = t_filteredTime.block(0,m_iFilterTaps/2+1,1,m_iWindowSize);
390  }
391 
392  //adds filtered channel to m_mapFilterChannels and m_allFilteredChannels
393  if(!m_mapFilteredChannels[type].contains(selected[i].row())) m_mapFilteredChannels[type].append(selected[i].row());
394  if(!m_allFilteredChannels.contains(selected[i].row())) m_allFilteredChannels.append(selected[i].row());
395 
396  //let RawModel emit a dataChanged() signal so that the views gets updated
397  QModelIndex changed = createIndex(selected[i].row(),selected[i].column());
398  emit dataChanged(changed,changed);
399 
400  qDebug() << "RawModel: Filtering applied to channel#" << selected[i].row();
401  }
402 
403  //Get name from TPassType's enum value
404  const QMetaObject& meta = ParksMcClellan::staticMetaObject;
405  int index = meta.indexOfEnumerator("TPassType");
406  QMetaEnum metaEnum = meta.enumerator(index);
407 
408  qDebug() << "RawModel: using PassType" << metaEnum.valueToKey(type);
409 }
