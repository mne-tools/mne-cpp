259  return;
260  }
261  else
262  {
263  printf("FiffStreamClient (assigned ID %d) accepted from\n\tIP:\t%s\n\tPort:\t%d\n\n",
264  m_iDataClientId,

273  {
274  //
275  // Write available data
276  //
277  if(m_qSendBlock.size() > 0)
278  {
279 // qDebug() << "is writeable " << t_qTcpSocket.isWritable();
280  qint32 t_iBlockSize = m_qSendBlock.size();
