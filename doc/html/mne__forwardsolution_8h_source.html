154  if(bUseWhitened)

156  t_kMeans.calculate(this->matRoiGWhitened, this->nClusters, p_RegionDataOut.roiIdx, p_RegionDataOut.ctrs, p_RegionDataOut.sumd, p_RegionDataOut.D);
157 
158  MatrixXd newCtrs = MatrixXd::Zero(p_RegionDataOut.ctrs.rows(), p_RegionDataOut.ctrs.cols());
159  for(qint32 c = 0; c < p_RegionDataOut.ctrs.rows(); ++c)
160  {
161  qint32 num = 0;
162 
163  for(qint32 idx = 0; idx < p_RegionDataOut.roiIdx.size(); ++idx)
164  {
165  if(c == p_RegionDataOut.roiIdx[idx])
166  {
167  newCtrs.row(c) += this->matRoiG.row(idx); //just take whitened to get indeces calculate centroids using the original matrix
168  ++num;
169  }
170  }
171 
172  if(num > 0)
173  newCtrs.row(c) /= num;
174  }
175  p_RegionDataOut.ctrs = newCtrs; //Replace whitened with original
176  }
177  else
178  t_kMeans.calculate(this->matRoiG, this->nClusters, p_RegionDataOut.roiIdx, p_RegionDataOut.ctrs, p_RegionDataOut.sumd, p_RegionDataOut.D);
179 
180  p_RegionDataOut.iLabelIdxOut = this->iLabelIdxIn;
